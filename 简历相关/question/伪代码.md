## 1. 核心链路与专项优化伪代码附录

### 1.1 设备接入伪代码

```ts
type ConnectStatus =
  | 'idle'
  | 'checking_permission'
  | 'connecting'
  | 'connected'
  | 'failed';

const connectDevice = async (input: DeviceInput) => {
  dispatch({ type: 'SET_STATUS', status: 'checking_permission' });

  const permission = await permissionService.checkDeviceConnectPermission();
  if (!permission.granted) {
    dispatch({
      type: 'CONNECT_FAILED',
      error: mapPermissionError(permission),
    });
    return;
  }

  dispatch({ type: 'SET_STATUS', status: 'connecting' });

  try {
    const device = await deviceService.bindDevice({
      sn: input.sn,
      wifi: input.wifi,
      source: input.source, // scan / manual / lan
    });

    dispatch({ type: 'CONNECT_SUCCESS', device });
    navigation.replace('DeviceDetail', { deviceId: device.id });
  } catch (error) {
    dispatch({
      type: 'CONNECT_FAILED',
      error: mapDeviceConnectError(error),
    });
  }
};
```

面试解释：

```text
设备接入链路不是页面里堆 if/else，而是把权限检查、连接中、成功、失败、重试抽成状态流转。页面只消费状态并展示对应 UI。
```

### 1.2 实时预览伪代码

原生预览桥接层可以抽象成下面这样。它的职责是统一 Android/iOS 对外暴露给 RN 的接口，底层再调用厂商/自研设备 SDK。

```ts
// RN 侧看到的是统一的 PreviewModule，而不是 Android/iOS SDK 细节。
interface NativePreviewModule {
  startPreview(params: {
    deviceId: string;
    streamType: 'main' | 'sub';
    quality: 'sd' | 'hd';
    muted: boolean;
  }): Promise<void>;

  stopPreview(params: { deviceId: string }): Promise<void>;

  switchQuality(params: {
    deviceId: string;
    quality: 'sd' | 'hd';
  }): Promise<void>;
}

type NativePreviewEventName =
  | 'onPlayerLoading'
  | 'onPlayerStarted'
  | 'onNetworkReconnect'
  | 'onPlayerError'
  | 'onPlayerStopped';

type NativePreviewEventPayload = {
  deviceId: string;
  code?: string;
  message?: string;
  connectionMode?: 'p2p' | 'relay' | 'lan';
};

const emitPreviewEventToRN = (
  eventName: NativePreviewEventName,
  payload: NativePreviewEventPayload,
) => {
  // Android 可以通过 DeviceEventManagerModule.RCTDeviceEventEmitter.emit 回传。
  // iOS 可以通过 RCTEventEmitter.sendEvent 回传。
  // RN 侧统一用 NativeEventEmitter 订阅这些事件。
  NativeBridgeEventEmitter.emit(eventName, payload);
};

// 原生桥接层内部可以这样调用自研设备 SDK。
// 这部分逻辑发生在 Android/iOS 原生侧，RN 侧只感知 PreviewModule 暴露的接口和事件。
const previewModuleNativeImpl = {
  async startPreview(params: StartPreviewParams) {
    emitPreviewEventToRN('onPlayerLoading', {
      deviceId: params.deviceId,
    });

    const authResult = await SelfDeviceSdk.authDevice({
      deviceId: params.deviceId,
    });

    if (!authResult.success) {
      emitPreviewEventToRN('onPlayerError', {
        deviceId: params.deviceId,
        code: 'DEVICE_AUTH_FAILED',
        message: '设备鉴权失败',
      });
      return;
    }

    const connection = await SelfDeviceSdk.createPreviewConnection({
      deviceId: params.deviceId,
      mode: 'p2p_first', // 优先 P2P，失败后由 SDK 内部降级 Relay / 局域网
    });

    await SelfDeviceSdk.startLiveStream({
      connectionId: connection.id,
      streamType: params.streamType,
      quality: params.quality,
      muted: params.muted,
      onStarted: () => {
        emitPreviewEventToRN('onPlayerStarted', {
          deviceId: params.deviceId,
          connectionMode: connection.mode,
        });
      },
      onReconnect: () => {
        emitPreviewEventToRN('onNetworkReconnect', {
          deviceId: params.deviceId,
          connectionMode: connection.mode,
        });
      },
      onError: sdkError => {
        emitPreviewEventToRN('onPlayerError', {
          deviceId: params.deviceId,
          code: mapSdkErrorCode(sdkError.code),
          message: mapSdkErrorMessage(sdkError.code),
        });
      },
    });
  },

  async stopPreview(deviceId: string) {
    await SelfDeviceSdk.stopLiveStream({ deviceId });
    await SelfDeviceSdk.releasePreviewConnection({ deviceId });

    emitPreviewEventToRN('onPlayerStopped', { deviceId });
  },
};

const nativePreviewBridge = {
  async startPreview(params: StartPreviewParams) {
    // 原生桥接层内部再调用厂商/自研设备 SDK：
    // 1. 校验设备鉴权
    // 2. 建立 P2P / Relay / 局域网连接
    // 3. 拉取对应码流
    // 4. 启动原生播放能力
    // 5. 通过事件把播放状态回传给 RN
    await NativeModules.PreviewModule.startPreview(params);
  },

  async stopPreview(deviceId: string) {
    // 停止播放、断开连接、释放原生资源。
    await NativeModules.PreviewModule.stopPreview({ deviceId });
  },
};
```

```ts
const usePreview = (params: PreviewParams) => {
  const [state, dispatch] = useReducer(previewReducer, initialPreviewState);

  const startPreview = async () => {
    dispatch({ type: 'PREVIEW_LOADING' });

    try {
      await previewService.startPreview(params);
    } catch (error) {
      dispatch({
        type: 'PREVIEW_ERROR',
        error: mapPreviewError(error),
      });
    }
  };

  useEffect(() => {
    const subscriptions = [
      previewEmitter.addListener('onPlayerLoading', event => {
        if (event.deviceId === params.deviceId) {
          dispatch({ type: 'PREVIEW_LOADING' });
        }
      }),
      previewEmitter.addListener('onPlayerStarted', event => {
        if (event.deviceId === params.deviceId) {
          dispatch({ type: 'PREVIEW_PLAYING' });
        }
      }),
      previewEmitter.addListener('onPlayerError', event => {
        if (event.deviceId === params.deviceId) {
          dispatch({
            type: 'PREVIEW_ERROR',
            error: mapPreviewError(event),
          });
        }
      }),
      previewEmitter.addListener('onNetworkReconnect', event => {
        if (event.deviceId === params.deviceId) {
          dispatch({ type: 'PREVIEW_RECONNECTING' });
        }
      }),
      previewEmitter.addListener('onPlayerStopped', event => {
        if (event.deviceId === params.deviceId) {
          dispatch({ type: 'PREVIEW_IDLE' });
        }
      }),
    ];

    startPreview();

    return () => {
      subscriptions.forEach(item => item.remove());
      previewService.stopPreview(params.deviceId);
    };
  }, [params.deviceId]);

  return {
    status: state.status,
    error: state.error,
    retry: startPreview,
  };
};
```

面试解释：

```text
实时预览链路的重点是页面不直接调原生模块，而是通过 previewService 和事件订阅，把原生播放事件映射成页面状态，并在退出时释放资源。
```

### 1.3 消息提醒伪代码

```ts
const useAlarmMessages = (deviceId?: string) => {
  const [messages, setMessages] = useState<AlarmMessage[]>([]);
  const [unreadCount, setUnreadCount] = useState(0);
  const [status, setStatus] = useState<'loading' | 'ready' | 'empty' | 'error'>(
    'loading',
  );

  const loadMessages = async (page = 1) => {
    setStatus('loading');

    try {
      const result = await messageService.getAlarmMessages({
        deviceId,
        page,
        pageSize: 20,
      });

      setMessages(result.list);
      setUnreadCount(result.unreadCount);
      setStatus(result.list.length > 0 ? 'ready' : 'empty');
    } catch (error) {
      setStatus('error');
    }
  };

  const openMessage = async (message: AlarmMessage) => {
    setMessages(list =>
      list.map(item =>
        item.id === message.id ? { ...item, read: true } : item,
      ),
    );

    await messageService.markAsRead(message.id);
    navigation.navigate('Playback', {
      deviceId: message.deviceId,
      recordTime: message.recordTime,
    });
  };

  return { messages, unreadCount, status, loadMessages, openMessage };
};
```

面试解释：

```text
消息提醒链路要把消息列表、未读数和页面状态拆开管理；点击消息后既要更新已读状态，也要跳转到对应设备或录像上下文。
```

### 1.4 录像回放伪代码

const usePlaybackList = (deviceId: string) => {
  const [filters, setFilters] = useState<PlaybackFilters>({
    date: today(),
    type: 'all',
  });
  const [records, setRecords] = useState<PlaybackRecord[]>([]);

      const result = await playbackService.getRecords({

      deviceId,
      recordId: record.id,
      startTime: record.startTime,
    });
  };

  return { filters, setFilters, records, status, loadRecords, playRecord };

### 1.5 多机型适配伪代码

```ts
import { Dimensions, Platform, PixelRatio } from 'react-native';
import { useSafeAreaInsets } from 'react-native-safe-area-context';

const guidelineWidth = 375;

const scaleSize = (size: number) => {
  const { width } = Dimensions.get('window');
  return Math.round((width / guidelineWidth) * size);
};

const normalizeFontSize = (size: number) => {
  const fontScale = PixelRatio.getFontScale();
  const scaledSize = scaleSize(size);

  // 字体缩放过大时做上限兜底，避免按钮、底部操作区被撑开。
  return fontScale > 1.2 ? Math.round(scaledSize / fontScale) : scaledSize;
};

const useDeviceAdaptiveLayout = () => {
  const insets = useSafeAreaInsets();
  const { width, height } = Dimensions.get('window');

  const isSmallScreen = width < 360;
  const isLandscape = width > height;

  return {
    pagePaddingHorizontal: isSmallScreen ? 12 : 16,
    bottomBarPaddingBottom: Math.max(insets.bottom, 12),
    previewHeight: isLandscape ? height * 0.72 : width * 0.56,
    titleFontSize: normalizeFontSize(isSmallScreen ? 15 : 16),
    toolbarStyle: Platform.select({
      ios: { paddingTop: insets.top },
      android: { elevation: 4 },
    }),
  };
};

const DevicePreviewPage = () => {
  const layout = useDeviceAdaptiveLayout();

  return (
    <Screen style={{ paddingHorizontal: layout.pagePaddingHorizontal }}>
      <PreviewPanel height={layout.previewHeight} />
      <BottomToolbar
        style={[
          layout.toolbarStyle,
          { paddingBottom: layout.bottomBarPaddingBottom },
        ]}
      />
    </Screen>
  );
};
```

面试解释：

```text
多机型适配不是每个页面到处写判断，而是把安全区、屏幕尺寸、字体缩放、平台差异收敛到 layout hook 或工具函数里，页面只消费计算后的布局参数。
```

### 1.6 长列表优化伪代码

```tsx
type RecordItemProps = {
  item: PlaybackRecord;
  onPress: (record: PlaybackRecord) => void;
};

const ITEM_HEIGHT = 88;
const PAGE_SIZE = 20;

// 示例场景：单设备录像回放列表。
// 普通设备单日几十到几百条录像片段，高频移动侦测设备可能达到 300-500 条。
// 前端不一次性渲染全部数据，而是按 20 条分页加载。
type PlaybackRecordPage = {
  list: PlaybackRecord[];
  nextCursor?: string;
  total?: number;
};

const RecordItem = React.memo(({ item, onPress }: RecordItemProps) => {
  const handlePress = useCallback(() => {
    onPress(item);
  }, [item, onPress]);

  return (
    <Pressable onPress={handlePress} style={{ height: ITEM_HEIGHT }}>
      <CachedImage source={{ uri: item.thumbnailUrl }} />
      <RecordMeta title={item.title} duration={item.duration} />
    </Pressable>
  );
});

const OptimizedRecordList = ({ deviceId }: { deviceId: string }) => {
  const [records, setRecords] = useState<PlaybackRecord[]>([]);
  const [cursor, setCursor] = useState<string | undefined>();
  const [loadingMore, setLoadingMore] = useState(false);

  const loadMore = useCallback(async () => {
    if (loadingMore) return;

    setLoadingMore(true);
    const result: PlaybackRecordPage = await playbackService.getRecords({
      deviceId,
      cursor,
      pageSize: PAGE_SIZE,
    });

    setRecords(prev => [...prev, ...result.list]);
    setCursor(result.nextCursor);
    setLoadingMore(false);
  }, [deviceId, cursor, loadingMore]);

  const openRecord = useCallback((record: PlaybackRecord) => {
    navigation.navigate('PlaybackPlayer', {
      deviceId,
      recordId: record.id,
      startTime: record.startTime,
    });
  }, [deviceId]);

  const renderItem = useCallback(
    ({ item }: { item: PlaybackRecord }) => (
      <RecordItem item={item} onPress={openRecord} />
    ),
    [openRecord],
  );

  const getItemLayout = useCallback(
    (_: unknown, index: number) => ({
      length: ITEM_HEIGHT,
      offset: ITEM_HEIGHT * index,
      index,
    }),
    [],
  );

  return (
    <FlatList
      data={records}
      keyExtractor={item => item.id}
      renderItem={renderItem}
      getItemLayout={getItemLayout}
      initialNumToRender={8}
      windowSize={5}
      maxToRenderPerBatch={8}
      updateCellsBatchingPeriod={50}
      removeClippedSubviews={Platform.OS === 'android'}
      onEndReached={loadMore}
      onEndReachedThreshold={0.4}
    />
  );
};
```

面试解释：

```text
长列表优化的核心是减少一次性渲染、减少重复渲染、减少 item 内计算。FlatList 负责虚拟化，React.memo/useCallback 保证 item 稳定，分页和缩略图缓存降低数据和图片压力。
```
